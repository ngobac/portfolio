Calculate NDVI from Satellite Imagery

Python script using GDAL/Rasterio to calculate Normalized Difference Vegetation Index (NDVI) from multispectral imagery.

import rasterio

import numpy as np

from rasterio.plot import show

import matplotlib.pyplot as plt

 

def calculate_ndvi(red_band_path, nir_band_path, output_path):

    """

    Calculate NDVI from red and NIR bands

    NDVI = (NIR - Red) / (NIR + Red)

    """

    # Read red and NIR bands

    with rasterio.open(red_band_path) as red:

        red_data = red.read(1).astype(float)

        profile = red.profile

 

    with rasterio.open(nir_band_path) as nir:

        nir_data = nir.read(1).astype(float)

 

    # Calculate NDVI

    ndvi = (nir_data - red_data) / (nir_data + red_data)

 

    # Handle division by zero

    ndvi = np.where((nir_data + red_data) == 0, 0, ndvi)

 

    # Save NDVI raster

    profile.update(dtype=rasterio.float32, count=1)

    with rasterio.open(output_path, 'w', **profile) as dst:

        dst.write(ndvi.astype(rasterio.float32), 1)

 

    print(f"NDVI calculated and saved to {output_path}")

    return ndvi

 

# Example usage

red_band = "landsat_band4.tif"

nir_band = "landsat_band5.tif"

output = "ndvi_output.tif"

 

ndvi_result = calculate_ndvi(red_band, nir_band, output)

Spatial Join and Analysis with GeoPandas

Perform spatial join operations and calculate area statistics using GeoPandas.

import geopandas as gpd

import pandas as pd

 

def spatial_join_analysis(polygons_path, points_path):

    """

    Perform spatial join and calculate statistics

    """

    # Read spatial data

    polygons = gpd.read_file(polygons_path)

    points = gpd.read_file(points_path)

 

    # Ensure same CRS

    points = points.to_crs(polygons.crs)

 

    # Spatial join: points within polygons

    joined = gpd.sjoin(points, polygons, how='left', predicate='within')

 

    # Calculate area in hectares

    polygons['area_ha'] = polygons.geometry.area / 10000

 

    # Group by polygon and count points

    stats = joined.groupby('index_right').size().reset_index(name='point_count')

 

    # Merge with polygon data

    result = polygons.merge(stats, left_index=True, right_on='index_right', how='left')

    result['point_count'].fillna(0, inplace=True)

 

    # Calculate density (points per hectare)

    result['density'] = result['point_count'] / result['area_ha']

 

    return result

 

# Example usage

forest_parcels = "data/forest_parcels.shp"

tree_points = "data/tree_locations.shp"

 

analysis_result = spatial_join_analysis(forest_parcels, tree_points)

print(analysis_result[['area_ha', 'point_count', 'density']].describe())

                `;

    });

 

// Color scale function

function getColor(density) {

    return density > 1000 ? '#064e3b' :

           density > 500  ? '#047857' :

           density > 200  ? '#10b981' :

           density > 100  ? '#34d399' :

                           '#a7f3d0';

}

PostGIS Spatial Queries

Advanced spatial queries using PostGIS for geospatial data analysis.

-- Find forest parcels within 500m of water sources

SELECT

    f.parcel_id,

    f.parcel_name,

    f.forest_type,

    ST_Area(f.geom) / 10000 AS area_hectares,

    MIN(ST_Distance(f.geom, w.geom)) AS distance_to_water

FROM

    forest_parcels f

CROSS JOIN

    water_sources w

WHERE

    ST_DWithin(f.geom, w.geom, 500)  -- Within 500 meters

    AND f.forest_type = 'Natural Forest'

GROUP BY

    f.parcel_id, f.parcel_name, f.forest_type, f.geom

HAVING

    ST_Area(f.geom) / 10000 > 10  -- Larger than 10 hectares

ORDER BY

    distance_to_water ASC;

 

-- Calculate buffer zones and intersection areas

WITH buffered_zones AS (

    SELECT

        parcel_id,

        ST_Buffer(geom, 100) AS buffer_geom  -- 100m buffer

    FROM

        forest_parcels

)

SELECT

    p.parcel_id,

    p.parcel_name,

    ST_Area(ST_Intersection(b.buffer_geom, p.geom)) / 10000 AS overlap_area_ha

FROM

    protected_areas p

INNER JOIN

    buffered_zones b ON ST_Intersects(p.geom, b.buffer_geom)

WHERE

    ST_Area(ST_Intersection(b.buffer_geom, p.geom)) > 0;

Spatial Aggregation and Statistics

Calculate spatial statistics and aggregations for forest management reporting.

-- Forest type distribution by province

SELECT

    p.province_name,

    f.forest_type,

    COUNT(*) AS parcel_count,

    SUM(ST_Area(f.geom)) / 10000 AS total_area_ha,

    AVG(ST_Area(f.geom)) / 10000 AS avg_parcel_size_ha,

    ST_Union(f.geom) AS combined_geometry

FROM

    forest_parcels f

INNER JOIN

    provinces p ON ST_Within(ST_Centroid(f.geom), p.geom)

GROUP BY

    p.province_name, f.forest_type

ORDER BY

    p.province_name, total_area_ha DESC;

 

-- Temporal change detection

WITH forest_2020 AS (

    SELECT parcel_id, ST_Area(geom) / 10000 AS area_ha

    FROM forest_parcels_2020

),

forest_2024 AS (

    SELECT parcel_id, ST_Area(geom) / 10000 AS area_ha

    FROM forest_parcels_2024

)

SELECT

    f20.parcel_id,

    f20.area_ha AS area_2020,

    f24.area_ha AS area_2024,

    (f24.area_ha - f20.area_ha) AS change_ha,

    ROUND(((f24.area_ha - f20.area_ha) / f20.area_ha * 100), 2) AS change_percent,

    CASE

        WHEN f24.area_ha > f20.area_ha THEN 'Gain'

        WHEN f24.area_ha < f20.area_ha THEN 'Loss'

        ELSE 'No Change'

    END AS change_type

FROM

    forest_2020 f20

LEFT JOIN

    forest_2024 f24 ON f20.parcel_id = f24.parcel_id

WHERE

    ABS(f24.area_ha - f20.area_ha) > 0.5  -- Changes > 0.5 ha

ORDER BY

    ABS(f24.area_ha - f20.area_ha) DESC;
